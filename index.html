drum_detection.py

# FastAPI endpoint for drum detection
@app.post("/api/detect-drums")
async def detect_drums(file: UploadFile):
    # Load and process audio file
    audio, sr = librosa.load(file.file, sr=44100)

    # Detect drum onsets using librosa
    onset_frames = librosa.onset.onset_detect(
        y=audio, sr=sr, threshold=0.7
    )
    onset_times = librosa.frames_to_time(onset_frames, sr=sr)

    # Classify drum types by frequency analysis
    drum_events = []
    for onset in onset_times:
        spectral_centroid = librosa.feature.spectral_centroid(
            y=audio, sr=sr
        )[0, 0]

        if spectral_centroid > 3000:
            drum_type = "hihat"
        elif spectral_centroid > 1000:
            drum_type = "snare"
        else:
            drum_type = "kick"

        drum_events.append({
            "time": onset,
            "type": drum_type
        })

    return {"drum_events": drum_events}

OTPController.php

<?php
// Laravel OTP Authentication Controller
class OTPController extends Controller
{
    public function sendOTP(Request $request)
    {
        $request->validate([
            'phone' => 'required|string|min:10',
            'method' => 'required|in:whatsapp,sms'
        ]);

        $otp = rand(100000, 999999);
        Cache::put("otp_{$request->phone}", $otp, 300);

        // Try WhatsApp first, fallback to SMS
        try {
            $this->sendWhatsAppOTP($request->phone, $otp);
        } catch (Exception $e) {
            SendSMSOTP::dispatch($request->phone, $otp);
        }

        return response()->json([
            'message' => 'OTP sent successfully'
        ]);
    }

    public function verifyOTP(Request $request)
    {
        $cachedOTP = Cache::get("otp_{$request->phone}");

        if ($cachedOTP !== $request->otp) {
            return response()->json(['error' => 'Invalid OTP'], 422);
        }

        $user = User::firstOrCreate(['phone' => $request->phone]);
        $token = $user->createToken('auth-token')->plainTextToken;

        return response()->json(['token' => $token]);
    }
}

fee-table-shortcode.php

<?php
// WordPress Dynamic Fee Table Shortcode
function maid_fee_table_shortcode($atts) {
    $atts = shortcode_atts([
        'countries' => 'all',
        'packages' => 'all',
        'show_filters' => 'true'
    ], $atts);

    // Get fee data from WordPress options
    $fee_data = get_option('maid_fee_table_data', []);

    // Filter data based on shortcode attributes
    $filtered_data = $this->filterFeeData($fee_data, $atts);

    ob_start();
    ?>
    <div class="maid-fee-table-container">
        <?php if ($atts['show_filters'] === 'true'): ?>
        <div class="fee-filters">
            <select id="country-filter">
                <option value="all">All Countries</option>
                <?php foreach ($filtered_data as $country => $packages): ?>
                    <option value="<?php echo esc_attr($country); ?>">
                        <?php echo esc_html($country); ?>
                    </option>
                <?php endforeach; ?>
            </select>
        </div>
        <?php endif; ?>

        <table class="maid-fee-table">
            <thead>
                <tr><th>Country</th><th>Package</th><th>Fee</th></tr>
            </thead>
            <tbody>
                <?php foreach ($filtered_data as $country => $packages): ?>
                    <?php foreach ($packages as $package => $fee): ?>
                    <tr>
                        <td><?php echo esc_html($country); ?></td>
                        <td><?php echo esc_html($package); ?></td>
                        <td>$<?php echo number_format($fee, 2); ?></td>
                    </tr>
                    <?php endforeach; ?>
                <?php endforeach; ?>
            </tbody>
        </table>
    </div>
    <?php
    return ob_get_clean();
}
add_shortcode('maid_fee_table', 'maid_fee_table_shortcode');
